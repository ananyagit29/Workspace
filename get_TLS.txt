package com.ipca.truckLoadStuff;

import com.ipca.common.Support;
import com.ipca.connection.DMSConnection;
import java.io.IOException;
import java.io.PrintWriter;
import java.sql.Connection;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;
import java.text.SimpleDateFormat;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import jakarta.servlet.http.HttpSession;

/**
 *
 * @author ajay.vadkar
 */
public class GetTruckLoadStuffServlet extends HttpServlet {

    /**
     *
     * @param request
     * @param response
     * @throws ServletException
     * @throws IOException
     */
    protected void processRequest(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doPost(request, response);
    }

    /**
     *
     * @param request
     * @param response
     * @throws ServletException
     * @throws IOException
     */
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doPost(request, response);
    }

    /**
     *
     * @param request
     * @param response
     * @throws ServletException
     * @throws IOException
     */
    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        PrintWriter out = response.getWriter();
        String op = request.getParameter("op");
        String page = request.getParameter("page");
        String tlStuffCode = request.getParameter("tlStuffCode");

        SimpleDateFormat sdf = new SimpleDateFormat("dd-MM-yyyy");
        HttpSession session = request.getSession();
        String uid = ((session.getAttribute("userid") != null) ? (String) session.getAttribute("userid") : "");
        String com = ((session.getAttribute("com") != null) ? (String) session.getAttribute("com") : "");
        String div = ((session.getAttribute("div") != null) ? (String) session.getAttribute("div") : "");
        String loc = ((session.getAttribute("loc") != null) ? (String) session.getAttribute("loc") : "");
        String app = ((session.getAttribute("app") != null) ? (String) session.getAttribute("app") : "");
        String finyear = ((session.getAttribute("finyear") != null) ? (String) session.getAttribute("finyear") : "");
        String fin_year = ((session.getAttribute("fin_year") != null) ? (String) session.getAttribute("fin_year") : "");
        String com_name = ((session.getAttribute("com_name") != null) ? (String) session.getAttribute("com_name") : "");
        String loc_name = ((session.getAttribute("loc_name") != null) ? (String) session.getAttribute("loc_name") : "");

        SimpleDateFormat format1 = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        SimpleDateFormat format2 = new SimpleDateFormat("dd-MM-yyyy HH:mm:ss");
        java.util.Date createdon = null;
        String createdOn = null;
        String Role = Support.getRole(uid, app, loc);
        Connection con = null, con1 = null;
        Statement st = null, st1 = null;
        try {
            con = DMSConnection.getNewInstance().getConnection();
            st = con.createStatement();

            String query = "select INVOICE_NO,CREATED_BY,CREATED_ON,FILE_NAME,file_path  "
                    + "from DMS_TRUCK_LOAD_STUFF  "
                    + "where FINANCIAL_YEAR = '" + finyear + "' "
                    + "and Location_Id='" + loc + "' "
                    + "and INVOICE_NO ='" + tlStuffCode + "' "
                    + "order by CREATED_ON ";
            ResultSet rs = st.executeQuery(query);
            int i = 0;
            if (rs.next()) {
                out.println("<div id='scrollacc'><table id='showData' name='showData' width='100%'>");
                out.println("<tr><th>Invoice Number</th><th>File Name</th>");
                if (Role.contains("Remove")) {
                    out.println("<th>Remove</th>");
                }
                out.println("<th>Created By</th><th>Created On</th>");
                do {
                    String invoice_no = rs.getString("INVOICE_NO");
                    createdon = format1.parse(rs.getString("created_on"));
                    createdOn = format2.format(createdon);
                    String filename = rs.getString("file_name") == null ? "" : rs.getString("file_name");
                    String filepath = rs.getString("File_Path") == null ? "" : rs.getString("File_Path");
                    String createdby = rs.getString("CREATED_BY");
                    if (filename != "") {
                        out.println("<tr><td>" + invoice_no + "</td>");
                        if (filepath == "") {
                            out.println("<td>" + filename + "</td>");
                            if (Role.contains("Remove")) {
                                out.println("<td style='width:100px;'>File Removed</td>");
                            }
                        } else {
                            String path = filepath.replaceAll(" ", "%20");
                            path = path.replaceAll("\\\\", "/");
                            out.println("<td><input type='button' id='buttonlink' onclick=\"callservlet('" + path + "')\" value='" + filename + "'/></td>");
                            if (Role.contains("Remove")) {
                                out.println("<td style='width:50px;'><button type='button' onclick=\"removeFile('" + tlStuffCode + "','" + filename + "','" + path + "')\"><img src='" + request.getContextPath() + "/images/remove.png' alt='Remove'></button></td>");
                            }
                        }
                        out.println("<td>" + createdby + "</td><td>" + createdOn + "</td></tr>");
                    }
                } while (rs.next());
                out.println("</table></div>");
            }

        } catch (Exception ex) {
            ex.printStackTrace();
            request.setAttribute("error", ex.getMessage());
        } finally {
            try {
                if (con != null) {
                    con.close();
                }
                if (st != null) {
                    st.close();
                }
                if (con1 != null) {
                    con1.close();
                }
                if (st1 != null) {
                    st1.close();
                }
            } catch (SQLException ex) {
            }
        }
    }

    /**
     *
     * @return
     */
    @Override
    public String getServletInfo() {
        return "Short description";
    }

}
