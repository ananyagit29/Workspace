                document.getElementById("tlStuffCode").value = tlStuffCode;

                var url = "<%=request.getContextPath()%>/GetTruckLoadStuffServlet?tlStuffCode=" + tlStuffCode;
                xmlHttpRequest.onreadystatechange = getlruckloadstuffdata;
                xmlHttpRequest.open("GET", url, true);
                xmlHttpRequest.send(null);

            }

                        }
                    }
                }
            }
