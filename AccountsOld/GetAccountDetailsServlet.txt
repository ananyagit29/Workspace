                        out.println("<div style='float:right'><input type='button' id='btndownload' value='Download' onclick='downloadFiles(" + i + ");'/></div>");
                    }
                } else {
                    out.println("<tr>");
                    out.println("<td><h1>Data Not Found</h1></td>");
                    out.println("</tr>");
                }
            } else if (op.equalsIgnoreCase("getpartydetails")) {
                ResultSet rs = st.executeQuery(query);
                out.println("<option value='SELECT'>SELECT</option>");
                while (rs.next()) {
                    String accName = rs.getString("ACCOUNT_CODE");
                    if (accName.equalsIgnoreCase(accAccName)) {
                        out.println("<option value='" + rs.getString("ACCOUNT_CODE") + "' selected>" + rs.getString("ACCOUNT_NAME") + "</option>");
                    } else {
                        out.println("<option value='" + rs.getString("ACCOUNT_CODE") + "'>" + rs.getString("ACCOUNT_NAME") + "</option>");
                    }
                }
            } else if (op.equalsIgnoreCase("getmissingdocuments")) {
                ResultSet rs = st.executeQuery(query);
                if (rs.next()) {
                    query = query.replaceAll("'", "%27");
                    query = query.replaceAll(" ", "%20");
                    query = query.replaceAll("-", "%2D");
                    query = query.replaceAll(":", "%3a");
                    query = query.replaceAll("_", "%5F");
                    query = query.replaceAll("\\+", "plus");
                    out.println("<tr><td><input type='hidden' id='accDaybookCode' name='accDaybookCode' value='" + accDaybookCode + "'/></td></tr>");
                    out.println("<tr><td><input type='hidden' id='accDaybookName' name='accDaybookName' value='" + accDaybookName + "'/></td></tr>");
                    out.println("<div style='float: right;'><input type='button' value='Save As Excel' onclick=\"window.open('" + request.getContextPath() + "/SaveAsExcel?filename=MissingDocumentsReport&query=" + query + "','_self')\"/></div>");
                    out.println("<br/><br/><div id='scrollacc' style='width: 1000px;'><table id='showData' name='showData' width='100%'>");
                    out.println("<tr><th>Document Number</th><th>Doc Date</th><th>FAS User</th></tr>");
                    do {
                        out.println("<tr><td>" + rs.getString("doc_code") + "</td>");
                        out.println("<td>" + sdf.format(rs.getDate("DOC_DATE")) + "</td>");
                        if (rs.getString("fas_user") == null) {
                            out.println("<td></td></tr>");
                        } else {
                            out.println("<td>" + rs.getString("fas_user") + "</td></tr>");
                        }
                    } while (rs.next());
                    out.println("</table></div>");
                } else {
                    out.println("<tr>");
                    out.println("<td><h1>No Missing Data Found</h1></td>");
                    out.println("</tr>");
                }
            }
        } catch (Exception ex) {
            request.setAttribute("error", ex.getMessage());
        } finally {
            try {
                if (con != null) {
                    con.close();
                }
                if (st != null) {
                    st.close();
                }
                if (con1 != null) {
                    con1.close();
                }
                if (st1 != null) {
                    st1.close();
                }
            } catch (SQLException ex) {
            }
        }
    }
