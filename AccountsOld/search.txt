            function resetMonth()
            {
                var year = '<%=finyear%>';
                var accDocMonth = document.getElementById("accDocMonth");
                accDocMonth.value = "04";
                let fiscalyear = "";
                let diff;
                let newmonth;
                let today = new Date();
                let month = today.getMonth() + 1;
                if ((month) <= 3) {
                    fiscalyear = (today.getFullYear() - 1) + "-" + today.getFullYear();
                    diff = 9;
                    newmonth = month + diff;
                } else {
                    fiscalyear = today.getFullYear() + "-" + (today.getFullYear() + 1);
                    diff = 3;
                    newmonth = month - diff;
                }
                if (year >= fiscalyear) {
                    var options = document.forms['formSearchAccounts']['accDocMonth'].options;
                    let len = options.length;
                    let i;
                    for (i = newmonth; i < len; i++) {
                        options[i].disabled = true;
                    }
                }
                getDocCodeDetails();
            }
