            function downloadFiles(i)
            {
                var checkboxes = document.querySelectorAll('input[type="checkbox"]');
                if (checkboxes.length <= 100)
                {
                    var filepathlist = "";
                    var arr = [];
                    var count = 0;
                    for (var j = 1; j < i; j++) {
                        if (checkboxes[j].checked)
                        {
                            count++;
                            document.getElementById('btndownload').disabled = true;
                            filepathlist = document.getElementById("row" + j).value;
                            arr.push(filepathlist.toString());
                        }
                    }
                    if (count === 0)
                    {
                        alert("Please select atleast one row to download");
                    } else
                    {
                        window.open('<%=request.getContextPath()%>/DownloadMultipleFilesServlet?filepathlist=' + arr, '_self');
                    }
                    document.getElementById('btndownload').disabled = false;
                } else
                {
                    alert("You can select only upto 100 rows");
                    document.getElementById('selectall').checked = false;
                }
            }

            function checkSelection(source) {
                var checkboxes = document.querySelectorAll('input[type="checkbox"]');
                if (checkboxes.length <= 100)
                {
                    for (var i = 0; i < checkboxes.length; i++) {
                        if (checkboxes[i] !== source)
                        {
                            checkboxes[i].checked = source.checked;
                        }
                    }
                } else
                {
                    alert("You can select only upto 100 rows");
                    document.getElementById('selectall').checked = false;
                }
            }
