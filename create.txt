            function doClick() {
                var options = document.getElementById("tlsSelectedInvoice").options;
                var tlsSelectedInvoice = document.getElementById("tlsSelectedInvoice");
                var tlsFile = document.getElementById("tlsFile");
                if (tlsSelectedInvoice !== null) {
                    if (tlsSelectedInvoice.length < 1) {
                        alert("Please select atleast one Invoice Number");
                    } else if (tlsFile.value === "")
                    {
                        alert("Please select File");
                    } else
                    {
                        var ext = tlsFile.value.split('.').pop();
                        if (ext === "pdf" || ext === "docx" || ext === "doc") {
                            for (var i = 0; i < options.length; i++) {
                                options[i].selected = true;
                            }
                            document.getElementById("formCreateTruckLoadStuff").submit();
                        } else
                        {
                            alert("Only Word or PDF format are allowed")
                            tlsFile.value = "";
                        }
                    }
                }
            }

            function moveInvoice(lstbxFrom, lstbxTo) {
                var varFromBox = document.getElementById(lstbxFrom);
                var varToBox = document.getElementById(lstbxTo);
                if ((varFromBox !== null) && (varToBox !== null)) {
                    if (varFromBox.length < 1) {
                        if (lstbxFrom === "tlsSelectInvoice")
                        {
                            alert('There are no items in the Select Invoice Number listbox.');
                        } else
                        {
                            alert('There are no items in the Selected Invoice Number listbox.');
                        }
                        return false;
                    }
                    if (varFromBox.options.selectedIndex === -1)
                    {
                        if (lstbxFrom === "tlsSelectInvoice")
                        {
                            alert('Please select an Invoice Number to add.');
                        } else
                        {
                            alert('Please select an Invoice Number to remove.');
                        }
                        return false;
                    }
                    while (varFromBox.options.selectedIndex >= 0) {
                        var newOption = new Option();
                        newOption.text = varFromBox.options[varFromBox.options.selectedIndex].text;
                        newOption.value = varFromBox.options[varFromBox.options.selectedIndex].value;
                        varToBox.options[varToBox.length] = newOption;
                        varFromBox.remove(varFromBox.options.selectedIndex);
                    }
                }
                return false;
            }
